# Tällä komennolla voit kopioida vaikka kokonaisia kansioita minionille komentorivillä. Komennon loppuosa varmistaa, että kansiosta tuhotaan aikaisemmat tiedostot ennenkuin kopioidaan uusia.
salt '*' file.copy /path/to/src_dir /path/to/dst_dir recurse=True remove_existing=True


# Laittaa minionin/minionit asentamaan paketin hallinnasta ohjelman.(toimii monilla eri linux distroilla= universaali komento)
salt '*' pkg.install vlc


# Kokeilee kaikille minioneille pingin ja listaa vastanneet minionit.
sudo salt '*' test.ping

# Listaa orjien verkkoosoitteet ja verkkolaitteet.('*'= meinaa että kaikille koneille. sevoi spesifioida esim = raspberry, salt-minion1)
sudo salt '*' network.interfaces

# Antaa minionin mac osoitteen.
sudo salt '*'network.hwaddr

# Tällä komennolla voit laittaa minionin suorittamaan komentorivi komentoja masterilta.
sudo salt '*' cmd.run 'tähän-komento-minkä-haluat-suorittaa'  (merkin '*' voi korvata erillisellä minionilla kutsumalla suoraan minionin hostname.)
